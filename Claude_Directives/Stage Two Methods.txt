  public sealed class ForecastContext

    public ForecastContext(int targetDrawIndex, int lookbackN)
    {
        TargetDrawIndex = targetDrawIndex;
        LookbackN = lookbackN;

        UpperBoundDrawIndex = targetDrawIndex - 1;
        LowerBoundDrawIndex = Math.Max(1, targetDrawIndex - lookbackN);
    }

   public double ComputeMovingCenter( ForecastContext ctx,int setNumber)
   {
       if (ctx.AllHistory == null || ctx.AllHistory.Count == 0)
           throw new InvalidOperationException("AllHistory is not initialized");

       var window = ctx.AllHistory
           .Where(x =>
               x.DrawIndex >= ctx.LowerBoundDrawIndex &&
               x.DrawIndex <= ctx.UpperBoundDrawIndex)
           .ToList();

       if (window.Count == 0)
           throw new InvalidOperationException("Empty sliding window");

       var values = window
           .Select(v => LottoHelper.GetValueFromSet(v, setNumber))
           .ToList();

       return values.Average();
   }

   public int ComputeSafeDelta( ForecastContext ctx, int setNumber, double percentile = 0.90)
   {
       if (ctx.AllHistory == null || ctx.AllHistory.Count == 0)
           throw new InvalidOperationException("AllHistory is not initialized");

       var window = ctx.AllHistory
           .Where(x =>
               x.DrawIndex >= ctx.LowerBoundDrawIndex &&
               x.DrawIndex <= ctx.UpperBoundDrawIndex)
           .ToList();

       if (window.Count == 0)
           throw new InvalidOperationException("Empty sliding window");

       double center = ComputeMovingCenter(ctx, setNumber);

       var deviations = window
           .Select(v => Math.Abs(
               LottoHelper.GetValueFromSet(v, setNumber) - center))
           .OrderBy(d => d)
           .ToList();

       int idx = (int)Math.Floor(percentile * (deviations.Count - 1));
       return (int)Math.Ceiling(deviations[idx]);
   }



   public int ComputeHotDelta( ForecastContext ctx, int setNumber, double percentile = 0.65)
   {
       if (ctx.AllHistory == null || ctx.AllHistory.Count == 0)
           throw new InvalidOperationException("AllHistory is not initialized");

       var window = ctx.AllHistory
           .Where(x =>
               x.DrawIndex >= ctx.LowerBoundDrawIndex &&
               x.DrawIndex <= ctx.UpperBoundDrawIndex)
           .ToList();

       if (window.Count == 0)
           throw new InvalidOperationException("Empty sliding window");

       double center = ComputeMovingCenter(ctx, setNumber);

       var deviations = window
           .Select(v => Math.Abs(
               LottoHelper.GetValueFromSet(v, setNumber) - center))
           .OrderBy(d => d)
           .ToList();

       int idx = (int)Math.Floor(percentile * (deviations.Count - 1));
       return (int)Math.Ceiling(deviations[idx]);
   }

   public (int min, int max) ComputeWindowMinMax( ForecastContext ctx, int setNbr)
   {
       var values = ctx.AllHistory
           .Where(x =>
               x.DrawIndex >= ctx.LowerBoundDrawIndex &&
               x.DrawIndex <= ctx.UpperBoundDrawIndex)
           .Select(x => LottoHelper.GetValueFromSet(x, setNbr))
           .ToList();

       if (values.Count == 0)
           throw new InvalidOperationException("Empty window");

       return (values.Min(), values.Max());
   }

       public void BackfillPredictions( string state, int daysBack, string modelVersion, trace tracer
)
       {
           const int LOOKBACK_DRAWS = 44;

           List<DateTime> drawDates = GetDrawDates(state, daysBack)
               .OrderBy(d => d)
               .ToList();

           int total = drawDates.Count;
           int i = 0;

           List<lottoVal> allHistory;
           using (lottoDal ld = new lottoDal())
           {
               allHistory = ld.LoadAllHistoryOnce(state);
           }

           using (lottoDal lda = new lottoDal())
           {
               foreach (DateTime forecastDate in drawDates)
               {
                   i++;

                   tracer?.Invoke(
                       $"Backfill {state} {i}/{total} : {forecastDate:yyyy-MM-dd}");

                   int asOfDrawIndex = lda.getDrawIndexBefore(state, forecastDate);
                   if (asOfDrawIndex <= 0)
                       continue;

                   ForecastEngine engine = new ForecastEngine(LOOKBACK_DRAWS);
                   ForecastContext ctx = engine.ProcessDrawingDate(asOfDrawIndex);
                   ctx.AllHistory = allHistory;

                   int[] safeMin = new int[7];
                   int[] safeMax = new int[7];
                   int?[] hotMin = new int?[7];
                   int?[] hotMax = new int?[7];

                   for (int setNumber = 1; setNumber <= 6; setNumber++)
                   {
                       double center = engine.ComputeMovingCenter(ctx, setNumber);

                       int safeDelta = engine.ComputeSafeDelta(ctx, setNumber);
                       int hotDelta = engine.ComputeHotDelta(ctx, setNumber);

                       int sLow = (int)Math.Round(center - safeDelta);
                       int sHigh = (int)Math.Round(center + safeDelta);
                       int hLow = (int)Math.Round(center - hotDelta);
                       int hHigh = (int)Math.Round(center + hotDelta);

                       var (minLegal, maxLegal) = engine.ComputeWindowMinMax(ctx, setNumber);

                       sLow = Math.Max(sLow, minLegal);
                       sHigh = Math.Min(sHigh, maxLegal);
                       hLow = Math.Max(hLow, minLegal);
                       hHigh = Math.Min(hHigh, maxLegal);

                       if (sLow > sHigh) sLow = sHigh;
                       if (hLow > hHigh) hLow = hHigh;

                       safeMin[setNumber] = sLow;
                       safeMax[setNumber] = sHigh;
                       hotMin[setNumber] = hLow;
                       hotMax[setNumber] = hHigh;
                   }

                   lda.PersistForecastBands(
                       state,
                       forecastDate,
                       modelVersion,
                       safeMin,
                       safeMax,
                       hotMin,
                       hotMax
                   );
               }
           }

           tracer?.Invoke("Backfill complete");
       }
	   
	   
	   private bool EnsureForecastCorridorsUpToDate()
{
    const int LOOKBACK_DRAWS = 44;
    const string MODEL_VERSION = "WF_v4_baseline";

    string state = StateCode;

    // 1) Get the latest draw date only
    DateTime latestDrawDate;
    using (lottoDal lda = new lottoDal())
    {
        latestDrawDate = lda.GetLastDrawingDate(state);
    }

    if (latestDrawDate == DateTime.MinValue)
        return true;

    // 2) Resolve draw index
    int drawIndex;
    using (lottoDal lda = new lottoDal())
    {
        drawIndex = lda.getDrawIndex(latestDrawDate, state);
    }

    if (drawIndex <= 0)
        return true;

    // 3) Skip if already forecasted
    using (lottoDal lda = new lottoDal())
    {
        if (lda.ForecastExists(state, latestDrawDate, MODEL_VERSION))
            return true;
    }

    // 4) Forecast engine + history
    ForecastEngine engine = new ForecastEngine(LOOKBACK_DRAWS);

    if (_allHistory == null)
    {
        using (lottoDal ld = new lottoDal())
        {
            _allHistory = ld.LoadAllHistoryOnce(state);
        }
    }

    ForecastContext ctx = engine.ProcessDrawingDate(drawIndex);
    ctx.AllHistory = _allHistory;

    // arrays indexed 1..6
    int[] safeMin = new int[7];
    int[] safeMax = new int[7];
    int?[] hotMin = new int?[7];
    int?[] hotMax = new int?[7];

    for (int setNumber = 1; setNumber <= 6; setNumber++)
    {
        double center = engine.ComputeMovingCenter(ctx, setNumber);
        int safeDelta = engine.ComputeSafeDelta(ctx, setNumber);
        int hotDelta = engine.ComputeHotDelta(ctx, setNumber);

        int sLow = (int)Math.Round(center - safeDelta);
        int sHigh = (int)Math.Round(center + safeDelta);
        int hLow = (int)Math.Round(center - hotDelta);
        int hHigh = (int)Math.Round(center + hotDelta);

        int[] limits = LotteryConfig.getSetLimits(setNumber, state);
        int minLegal = limits[0];
        int maxLegal = limits[1];

        safeMin[setNumber] = Math.Max(sLow, minLegal);
        safeMax[setNumber] = Math.Min(sHigh, maxLegal);

        hotMin[setNumber] = Math.Max(hLow, minLegal);
        hotMax[setNumber] = Math.Min(hHigh, maxLegal);
    }

    // 5) Persist ONE new forecast date
    using (lottoDal lda = new lottoDal())
    {
        lda.PersistForecastBands(
            state,
            latestDrawDate,
            MODEL_VERSION,
            safeMin,
            safeMax,
            hotMin,
            hotMax
        );
    }

    return true;
}
